## 3. Баланс даши

### 3.1 Алгоритм

```
ВХОД: moon_longitude (0.0-360.0°, сидерическая)

nakshatra_index = floor(moon_longitude / 13.333333)        # 0-based
lord = VIMSHOTTARI_SEQ[nakshatra_index % 9]
nakshatra_start = nakshatra_index * 13.333333
remaining_deg = 13.333333 - (moon_longitude - nakshatra_start)
balance_ratio = remaining_deg / 13.333333
balance_years = MAHADASHA_YEARS[lord] * balance_ratio

# Перевод:
full_years = floor(balance_years)
remaining_months = (balance_years - full_years) * 12
full_months = floor(remaining_months)
full_days = round((remaining_months - full_months) * 30)

ВЫХОД: lord, full_years, full_months, full_days
```

### 3.2 Высокоточная формула (в минутах дуги)

```
moon_minutes = moon_longitude * 60
nakshatra_index = floor(moon_minutes / 800)
remaining_minutes = 800 - (moon_minutes - nakshatra_index * 800)
balance_ratio = remaining_minutes / 800
balance_days = MAHADASHA_YEARS[lord] * 365.25 * balance_ratio
```

### 4.3 Программная генерация

```python
def get_antardashas(md_lord, md_start, md_end):
    SEQUENCE = ['Ketu','Venus','Sun','Moon','Mars','Rahu','Jupiter','Saturn','Mercury']
    YEARS = {'Ketu':7,'Venus':20,'Sun':6,'Moon':10,'Mars':7,'Rahu':18,'Jupiter':16,'Saturn':19,'Mercury':17}
    idx = SEQUENCE.index(md_lord)
    ad_seq = SEQUENCE[idx:] + SEQUENCE[:idx]
    md_years = YEARS[md_lord]
    current = md_start
    result = []
    for planet in ad_seq:
        days = (md_years * YEARS[planet] / 120) * 365.25
        end = current + timedelta(days=days)
        result.append({'planet': planet, 'start': current, 'end': end, 'days': round(days)})
        current = end
    return result
```

## 9. Верификация

```python
def test_nakshatra_lookup():
    """Тест: 185.0° → Chitra (#14), lord=Mars"""
    idx = int(185.0 // 13.333333)  # 13
    assert idx + 1 == 14  # Chitra
    assert VIMSHOTTARI_SEQ[idx % 9] == 'Mars'

def test_dasha_balance():
    """Тест: moon=185.0° → Mars 0.875 лет = 10м 15д"""
    idx = 13
    start = idx * 13.333333  # 173.333°
    remaining = 13.333333 - (185.0 - start)  # 1.6667°
    ratio = remaining / 13.333333  # 0.125
    balance = 7 * ratio  # 0.875
    assert abs(balance - 0.875) < 0.001

def test_antardasha():
    """Тест: Ve-Ve = 1217д, Sa-Me = 983д"""
    assert round((20 * 20 / 120) * 365.25) == 1217
    assert round((19 * 17 / 120) * 365.25) == 983

def test_navamsha():
    """Тест: Ashwini P1→Овен, Bharani P1→Лев, Revati P4→Рыбы"""
    assert (0*4+0) % 12 == 0   # Овен
    assert (1*4+0) % 12 == 4   # Лев
    assert (26*4+3) % 12 == 11 # Рыбы

def test_triple_guna():
    """Тест: #1→Ra-Ra-Ra, #27→Sa-Sa-Sa"""
    G = ['Rajas','Tamas','Sattva']
    # n=1
    assert G[0//9] == 'Rajas' and G[0//3] == 'Rajas' and G[0%3] == 'Rajas'
    # n=27
    assert G[26//9] == 'Sattva' and G[(26%9)//3] == 'Sattva' and G[26%3] == 'Sattva'
```
